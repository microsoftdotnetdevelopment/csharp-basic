a do while loop is a control flow statement that executes a block of code at least once, and then repeatedly executes the block, or not, 
depending on a given boolean condition at the end of the block.
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication9
{
    class Program
    {
        static void Main(string[] args)
        {
            int ch;
            do
            {
                Console.WriteLine("enter a no.");
                int num = int.Parse(Console.ReadLine());
                int aq = num * num;
                Console.WriteLine("Square is "+ aq);
                Console.WriteLine("\n enter 1 to continue ");
                ch = int.Parse(Console.ReadLine());
            }
            while (ch == 1);
            Console.ReadKey();
        }
    }
}
