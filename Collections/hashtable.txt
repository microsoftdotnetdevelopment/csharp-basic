The Hashtable class represents a collection of key-and-value pairs that are organized based on the hash code of the key.
 It uses the key to access the elements in the collection.
class Program
    {
        static void Main(string[] args)
        {
            // Creates and initializes a new Hashtable.
            Hashtable htObj = new Hashtable();

            htObj.Add(1,"A");

            htObj.Add(2,"B");

            htObj.Add(3, "C");

            // Displays the properties and values of the ArrayList.
            Console.WriteLine("Count:    {0}", htObj.Count);

            Console.WriteLine();
            Console.WriteLine("Values");

            foreach (DictionaryEntry item in htObj)
            {
                Console.WriteLine("Key : {0}, Value: {1}", item.Key, item.Value);
            }

            //result will be null as key is not in hashtable
            string result = htObj[4] as string;

            Console.WriteLine("result is null :{0}", string.IsNullOrEmpty(result));
            Console.Read();

        }
    }
